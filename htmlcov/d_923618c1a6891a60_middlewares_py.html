20def categories4menu(request): 

21 """Добавляем контекст - меню, количество, категории итд.""" 

22 # сортируем выдачу работ по порядку заданному полем order 

23 # order задаем рандомно на 24 часа, потом снова сортируем 

24 if not cache.get('seed'): 

25 # если в кэше нет seed 

26 # (прошли сутки или перегружался сайт или еще что-то случилось) 

27 # сохраняем seed в кэш 

28 seed = datetime.date.today().strftime('%Y%m%d') 

29 cache.set('seed', seed, 60*60*24) 

30 # задаем новый порядок сортировки и новые тумбы если есть 

31 # на следующие сутки 

32 random.seed(int(seed)) 

33 prod = Product.objects.all().only('order', 'thumb_of_day') 

34 for p in prod: 

35 p.order = random.randint(1, 10000) 

36 p.thumb_of_day = p.thumbnail 

37 Product.objects.bulk_update(prod, fields=['order', 'thumb_of_day']) 

38 

39 context = {} 

40 

41 context['query'] = request.GET.get('query', '') 

42 context['searchform'] = SearchForm(initial={'query': context['query']}) 

43 

44 context['total_obj'] = Product.objects.count() 

45 context['total_cat'] = Category.objects.count() 

46 context['total_tec'] = Technique.objects.count() 

47 context['total_aut'] = Author.objects.count() 

48 

49 context['all_items_cat'] = Category.objects.all().\ 

50 annotate(num_arts=Count('category_products')).order_by('title') 

51 context['all_items_tec'] = Technique.objects.all().\ 

52 annotate(num_arts=Count('technique_products')).order_by('title') 

53 context['all_items_aut'] = Author.objects.all().\ 

54 annotate(num_arts=Count('author_products')).\ 

55 annotate(title=Concat(F('first_name'), Value(' '), F('last_name'))).\ 

56 only('id', 'last_name', 'first_name', 'slug',).order_by('title') 

57 

58 context['cart'] = Cart(request) 

59 context['STRIPE_PK'] = settings.STRIPE_PK 

60 

61 return context
