8class IndexView(ListView):

11 paginate_by = 9 

12 page_title = ''

15 context = super().get_context_data(**kwargs)

19 def get_queryset(self): 

20 self.page_title = '' 

21 slug = self.kwargs.get('slug', None) 

22 if slug and slug not in 'all': 

23 # выбираем все продукты по фильтру 

24 # и таблицу для определения названия категории 

25 mod, q = { 

26 'cat': [Category, Q(category__slug=slug)], 

27 'tec': [Technique, Q(technique__slug=slug)], 

28 'aut': [Author, Q(author__slug=slug)], 

29 }.get(self.kwargs.get('filter', None), Q()) 

30 self.page_title = mod.objects.get(slug=slug).__str__ 

31 else: 

32 query = self.request.GET.get('query', None) 

33 if query: 

34 # запрос в модель на поиск 

35 # в SQLite не работает регистронезависимый поиск 

36 # в MySQL все работает норм 

37 q = Q(title__icontains=query) |\ 

38 Q(description__icontains=query) |\ 

39 Q(materials__icontains=query) |\ 

40 Q(category__title__icontains=query) |\ 

41 Q(author__last_name__icontains=query) |\ 

42 Q(technique__title__icontains=query) 

43 elif self.request.path == '/favorites/': 

44 self.page_title = 'Избранное' 

45 cart = Cart(self.request) 

46 product_ids = cart.cart.keys() 

47 q = Q(pk__in=product_ids) 

48 else: 

49 # выбираем все продукты 

50 q = Q() 

51 

52 return Product.objects.filter(q).only( 

53 'order', 'title', 'slug', 'created', 'description', 'thumb_of_day', 

54 ).order_by('order', '-created')
